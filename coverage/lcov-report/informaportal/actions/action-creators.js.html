export function fetchWeather(city) {
  if (!city) {
    city = 'New York';
  }
 
  const url = `${ROOT_URL_WEATHER}&q=${city}&days=5`;
  const request = axios.get(url);
 
  console.log('request:', request);
 
  return {
    type: FETCH_WEATHER,
    payload: request,
  };
}
